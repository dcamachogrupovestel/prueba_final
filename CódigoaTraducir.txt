//Traducir a python
using System;

public class Persona
{
    public string Nombre { get; set; }
    public int Edad { get; set; }
    public string Pais { get; set; }

    public Persona(string nombre, int edad, string pais)
    {
        Nombre = nombre;
        Edad = edad;
        Pais = pais;
    }

    public void ImprimirNombre()
    {
        Console.WriteLine("El nombre de la persona es: " + Nombre);
    }

    public void ImprimirEdad()
    {
        Console.WriteLine("La edad de la persona es: " + Edad);
    }

    public void ImprimirPais()
    {
        Console.WriteLine("El pais de origen de la persona es: " + Pais);
    }

    public void ImprimirDatos()
    {
        Console.WriteLine("Nombre: " + Nombre);
        Console.WriteLine("Edad: " + Edad);
        Console.WriteLine("Pais de origen: " + Pais);
    }
}

public class Estudiante : Persona
{
    public Estudiante(string nombre, int edad, string pais) : base(nombre, edad, pais) { }

    public void ImprimirNombre()
    {
        Console.WriteLine("El nombre del estudiante es: " + Nombre);
    }

    public void ImprimirEdad()
    {
        Console.WriteLine("La edad del estudiante es: " + Edad);
    }

    public void ImprimirPais()
    {
        Console.WriteLine("El pais de origen del estudiante es: " + Pais);
    }
}

public class Profesor : Persona
{
    public Profesor(string nombre, int edad, string pais) : base(nombre, edad, pais) { }

    public void ImprimirNombre()
    {
        Console.WriteLine("El nombre del profesor es: " + Nombre);
    }

    public void ImprimirEdad()
    {
        Console.WriteLine("La edad del profesor es: " + Edad);
    }

    public void ImprimirPais()
    {
        Console.WriteLine("El pais de origen del profesor es: " + Pais);
    }
}

public class Curso
{
    public string Nombre { get; set; }
    public string Duracion { get; set; }
    public Profesor Profesor { get; set; }
    public Estudiante Estudiante { get; set; }

    public Curso(string nombre, string duracion, Profesor profesor, Estudiante estudiante)
    {
        Nombre = nombre;
        Duracion = duracion;
        Profesor = profesor;
        Estudiante = estudiante;
    }

    public void ImprimirNombre()
    {
        Console.WriteLine("El nombre del curso es: " + Nombre);
    }

    public void ImprimirDuracion()
    {
        Console.WriteLine("La duracion del curso es: " + Duracion);
    }

    public void ImprimirProfesor()
    {
        Console.WriteLine("El profesor del curso es: " + Profesor.Nombre + ", tiene la edad de " + Profesor.Edad + " años y proviene de " + Profesor.Pais);
    }

    public void ImprimirEstudiantes()
    {
        Console.WriteLine("Los estudiantes del curso son: " + Estudiante.Nombre + ", tiene la edad de " + Estudiante.Edad + " años y proviene de " + Estudiante.Pais);
    }
}
